For Developers

5 lines to give your AI superpowers

# Add ILUMINATY to Claude Code
claude mcp add iluminaty -- python -m iluminaty.mcp

# Then just ask Claude:
"What do you see on my screen?"
"Click the Save button"
"Run npm test and tell me what failed"
from iluminaty_client import Iluminaty

eye = Iluminaty()
snapshot = eye.see()              # see the screen
eye.post("/agent/do",            # natural language action
    instruction="save the file")
eye.post("/terminal/exec",        # run commands
    cmd="npm test")
# See what's on screen
curl localhost:8420/vision/snapshot

# Execute an action (natural language)
curl -X POST "localhost:8420/agent/do?instruction=save+the+file"

# Click a UI element by name
curl -X POST "localhost:8420/ui/click?name=Submit"
